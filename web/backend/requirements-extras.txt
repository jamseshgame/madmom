# Packages whose declared metadata constraints we need to bypass via
# `pip install --no-deps -r requirements-extras.txt`. Install these AFTER
# the main requirements.txt so the actual torch/numpy already on disk
# remain untouched.
#
# Why each entry needs the bypass:
#
# - chatterbox-tts 0.1.7: declares torch==2.6.0 and numpy<2.0.0, which
#   conflicts with the lockstep trio in requirements.txt (torch 2.12 +
#   numpy 2). Runtime works fine on the newer trio — Resemble's pin is
#   defensive. Lazy-loaded on first /api/tts/synth call.
chatterbox-tts==0.1.7

# - allin1: mir-aidj/all-in-one joint beat+downbeat+tempo+segment model.
#   May not install cleanly on Windows due to native deps; engine is
#   guarded with try/except import, tests use pytest.importorskip.
allin1>=1.0.0

# - basic-pitch: Spotify's polyphonic note transcription model.
#   May not install cleanly on Windows due to TensorFlow native deps.
#   Lazy-loaded on first use; tests use pytest.importorskip.
#   `pretty_midi` is a runtime dep of basic-pitch.predict that pip doesn't
#   pull in automatically — listing it explicitly here so a fresh deploy
#   doesn't crash on the first basic-pitch call.
basic-pitch>=0.3.0
pretty_midi>=0.2.10

# - aubio: C-backed audio processing library for fast onset detection.
#   May not install cleanly on Windows due to C/Cython native deps.
#   Engine is guarded with try/except import; function lazy-loads aubio.
aubio>=0.4.9
